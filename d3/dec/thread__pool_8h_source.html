47 class thread_pool_impl {
48 public:
49 thread_pool_impl(
50 const unsigned int &number_of_threads = std::thread::hardware_concurrency()) {
51 for (std::size_t i = 0; i < number_of_threads; ++i) {
52 queues_.push_back(std::make_unique<task_pair>());
53 threads_.emplace_back([&, id = i](std::stop_token stop_tok) {
54 do {
55 // check if we have task
56 if (queues_[id]->tasks.empty()) {
57 // no tasks, so we wait instead of spinning
58 queues_[id]->semaphore.acquire();
59 }
60
61 // ensure we have a task before getting task
62 // since the dtor releases the semaphore as well
63 if (!queues_[id]->tasks.empty()) {
64 // get the task
65 auto &task = queues_[id]->tasks.front();
66 // invoke the task
67 std::invoke(std::move(task));
68 // remove task from the queue
69 queues_[id]->tasks.pop();
70 }
71 } while (!stop_tok.stop_requested());
72 });
73 }
74 }

76 ~thread_pool_impl() {
77 // stop all threads

81 threads_[i].join();
82 }

94 template <typename Function, typename... Args,
95 typename ReturnType = std::invoke_result_t<Function &&, Args &&...>>
96 requires std::invocable<Function, Args...>
97 [[nodiscard]] std::future<ReturnType> enqueue(Function &&f, Args &&...args) {
98 // use shared promise here so that we don't break the promise later
99 auto shared_promise = std::make_shared<std::promise<ReturnType>>();
100 // here we move arguments into a tuple since we cannot do an init-capture followed by an
101 // ellipsis doing arguments = std::move(args)... would be better here, but currently not
102 // possible. see: http://eel.is/c++draft/expr.prim.lambda#capture-17.sentence-2 see:
103 // http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2017/p0780r1.html
104 auto task = [func = std::move(f), tup = std::make_tuple(std::move(args)...),
105 promise = shared_promise]() { promise->set_value(std::apply(func, tup)); };
106 // get the future before enqueuing the task
107 auto future = shared_promise->get_future();
108 // enqueue the task
109 enqueue_task(std::move(task));
110 return future;
111 }
